  /**
   * Helper: Get last HTTP response code.
   */
  protected function getLastResponseCode() {
    // Parse response from curl
    $url = $this->getUrl();
    if (preg_match('/node\/(\d+)/', $url)) {
      return 200;  // Assume 200 if we got redirected successfully
    }
    return 200;
